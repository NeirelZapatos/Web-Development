How to create a table that has a foreign key

CREATE TABLE orders ( id INT NOT NULL, order_number INT, product_id INT, PRIMARY KEY (id), FOREIGN KEY (product_id) REFERENCES products(id) )

Inner Join TABLE

SELECT orders.order_number, products.name, products.price FROM orders INNER JOIN products ON orders.product_id = products.id
