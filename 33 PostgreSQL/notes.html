SELECT * FROM (TABLE);

SELECT (COLUMN) FROM (TABLE);

SELECT (COLUMN), (COLUMN) FROM (TABLE);

SELECT (COLUMN) FROM (TABLE) WHERE (CONDITION)

Ex: SELECT rice FROM world_food WHERE country = 'United States';

SELECT (COLUMN) FROM (TABLE) WHERE (COLUMN) LIKE (PATTERN)

Ex: SELECT country FROM world_food WHERE country LIKE 'U' || '%';

This shows all the counties that start with the letter U

Ex 2: SELECT country FROM world_food WHERE country LIKE '%' || 'a';
